Employee.Emp_Brithday, Employee.Emp_Age, Employee.Emp_Tel , case depname when isnull then ‘ไม่มีแผนก’ else department.depname end

3.จงหาข้อมูลหัวหน้าแผนกแต่ละแผนก
Select  Employee.Empid, Employee .First_Empname, Employee.Last_Empname,Employee.Emp_Brithday, Employee.Emp_Age, Employee.Emp_Tel, .Level.Levname
From Employee, Level
Where Employee.Empid = Level.Empid and Level.Levname = ‘Managerdepartment’;
